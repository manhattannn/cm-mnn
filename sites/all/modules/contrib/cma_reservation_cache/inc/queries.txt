simple:
SELECT cache_id, cache_time, content_type, content_type_machine_name, bucket_slot_id, cache_slot_start_time, cache_slot_end_time, cache_slot_status, inventory, inventory_name, bucket_category, bucket_category_name, bucket_weight, item_nid, item_title, item_status, reservation_nid, reservation_group_id, reservation_uid, reservation_user_name, reservation_status, reservation_status_name, project_nid, project_title, field_reservations_date_start, field_reservations_date_end, field_reservation_id, field_reservation_notes_internal, field_reservation_notes_producer, field_penalty_points, field_temp_project_id, field_temp_facility_code, field_temp_inventory_id, field_temp_reserved_items, og_node1, field_reservation_cmptd_producer, field_override_reservation_email  FROM cma_reservation_cache  WHERE  cache_slot_start_time >= 1540180800 AND  cache_slot_start_time < 1540267200  AND  ( inventory = 1680  )  ORDER BY  bucket_weight, bucket_category_name, content_type, if(bucket_slot_id>0, bucket_slot_id, item_nid), cache_slot_start_time  LIMIT 100000

SELECT cache_id, cache_time, content_type, content_type_machine_name, bucket_slot_id, cache_slot_start_time, cache_slot_end_time, cache_slot_status, inventory, inventory_name, bucket_category, bucket_category_name, bucket_weight, item_nid, item_title, item_status, reservation_nid, reservation_group_id, reservation_uid, reservation_user_name, reservation_status, reservation_status_name, project_nid, project_title, field_reservations_date_start, field_reservations_date_end, field_reservation_id, field_reservation_notes_internal, field_reservation_notes_producer, field_penalty_points, field_temp_project_id, field_temp_facility_code, field_temp_inventory_id, field_temp_reserved_items, og_node1, field_reservation_cmptd_producer, field_override_reservation_email  FROM cma_reservation_cache  WHERE  cache_slot_start_time >= 1539748800 AND  cache_slot_start_time < 1539835200  AND  hour(from_unixtime(cache_slot_start_time + -14400)) >= 12  AND  hour(from_unixtime(cache_slot_start_time + -14400)) < 18 AND  ( inventory = 1680  )  AND  ( bucket_category = 931  OR  bucket_category = 857  )  AND  ( content_type_machine_name = '59th_conference_rooms_fac_'  OR  content_type_machine_name = '59th_open_closed_studios_fac_'  )  AND  ( item_nid = 36776  OR  item_nid = 374849  )  AND  dayofweek(from_unixtime(cache_slot_start_time + -14400)) = 4 ORDER BY  bucket_weight, bucket_category_name, content_type, if(bucket_slot_id>0, bucket_slot_id, item_nid), cache_slot_start_time  LIMIT 100000

QUERY #2 Gets Reservable Resoure and Bucket Items
All Items:
SELECT n.nid, n.title, n.type, rin.reservations_default_availability
     FROM node n
          INNER JOIN reservations_reservation_item_node rin ON n.vid = rin.vid WHERE rin.reservations_sub_type = 1 ORDER BY n.type ASC, n.nid DESC

One Content Type:
SELECT n.nid, n.title, n.type, rin.reservations_default_availability
     FROM node n
          INNER JOIN reservations_reservation_item_node rin ON n.vid = rin.vid WHERE rin.reservations_sub_type = 1  AND (n.type = 'audio_mixer_cm_' ) ORDER BY n.type ASC, n.nid DESC
